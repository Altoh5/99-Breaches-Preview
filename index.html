        // Toggle Chapter Accordion

        function toggleChapter(num) {

            const content = document.getElementById(`content-chapter-${num}`);

            const isExpanded = content.classList.contains('expanded');

            

            // Auto-close others for a cleaner UX

            document.querySelectorAll('.chapter-content').forEach(el => {

                el.classList.remove('expanded');

            });


            if (!isExpanded) {

                content.classList.add('expanded');

            }

        }


        // Global Search Functionality

        searchInput.addEventListener('input', (e) => {

            const term = e.target.value.toLowerCase();

            

            if (term === '') {

                if(currentSectionId) selectSection(currentSectionId);

                return;

            }


            // If searching while on About page, switch header context back

            document.getElementById('content-header').style.display = 'block';


            // Flatten all chapters to search through everything

            let results = [];

            bookData.forEach(section => {

                section.chapters.forEach(chap => {

                    if (

                        chap.title.toLowerCase().includes(term) || 

                        chap.context.toLowerCase().includes(term) || 

                        chap.learnings.toLowerCase().includes(term)

                    ) {

                        results.push({...chap, sectionName: section.shortTitle});

                    }

                });

            });


            // Update UI for Search Results

            document.getElementById('current-section-badge').textContent = 'Global Search';

            document.getElementById('current-section-title').textContent = `Found ${results.length} cases`;

            document.getElementById('current-section-desc').textContent = `Displaying search results for "${term}" across all training modules.`;

            

            // Remove active state from sidebar

            currentSectionId = null;

            renderSidebar();


            renderChapters(results);

            

            // Auto-expand all search results for easier scanning

            results.forEach(r => {

                const el = document.getElementById(`content-chapter-${r.num}`);

                if(el) el.classList.add('expanded');

            });

        });
